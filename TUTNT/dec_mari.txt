ACTOR BerserkMarine : BaseAllyMarine 30000
{
	//$Angled
	//$Color 8
	//$Category "TUTNT/Allied Marines"
	Tag "Space Marine [Berserked]"
	
	MeleeDamage 25
	MeleeSound "marine/punch"
	TRANSLATION 2
	Obituary "%o got beaten up by a berserker marine"
	States
	{
		Spawn:
			ALY1 AB 10 A_Look
			Loop
		See:
			ALY1 A 0
			{
				user_shotsfired == 0;
				user_nolegendarytrans == 1;
			}
			ALY1 AAAABBBBCCCCDDDD 1 
			{
				A_Chase;
			}
			Loop
		Melee:
			TNT1 A 0
			ALY1 D 4 A_FaceTarget
			ALY1 E 13 A_MeleeAttack
			Goto See
		Pain:
			ALY1 G 4
			ALY1 G 4 A_Pain
			Goto See
		Death:
			ALY1 H 10
			ALY1 I 10 A_Scream
			ALY1 J 10 A_Fall
			ALY1 KLM 10
			ALY1 M -1
			Stop
		XDeath:
			ALY1 O 5
			ALY1 P 5 A_XScream
			ALY1 Q 5 A_Fall
			ALY1 RSTUV 5
			ALY1 W -1
			Stop
	}
}
