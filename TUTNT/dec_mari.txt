ACTOR BerserkMarine : BaseAllyMarine 30000

	 A_Chase;
	}
	Loop
  Melee:
    TNT1 A 0 ACS_NamedExecuteAlways("RefreshStealthInfo",0,1) // makes a Stealth Monster become more visible
    ALY1 D 4 A_FaceTarget
	ALY1 E 13 A_MeleeAttack
	Goto See
  Pain:
    ALY1 G 4
	ALY1 G 4 A_Pain
	Goto See
  Death:
    ALY1 H 10
	ALY1 I 10 A_Scream
	ALY1 J 10 A_Fall

	ALY1 M -1
	Stop
  XDeath:
    ALY1 O 5

  }
}
